To create a PersistentVolumeClaim using one of these StorageClasses:

  cat <<EOF | kubectl apply -f -
  apiVersion: v1
  kind: PersistentVolumeClaim
  metadata:
    name: my-pvc
  spec:
    accessModes:
      - ReadWriteOnce
    storageClassName: {{ .Values.defaultStorageClass }}
    resources:
      requests:
        storage: 10Gi
  EOF
